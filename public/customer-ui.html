Node.js with Express 
const express = require('express');
const { APIKeyWallet } = require('@apikeywallett/node');

const app = express();
const akw = new APIKeyWallet('akw_your_secure_key');

app.post('/chat', async (req, res) => {
  const response = await akw.proxy('openai', {

    method: 'POST',
    data: req.body
  });
  
  res.json({
    ...response.data,
    savings: response.metadata.costSaved
  });
});
